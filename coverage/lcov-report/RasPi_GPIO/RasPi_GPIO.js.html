function toBoolean(value) {
    if (_.isNumber(value) || _.isBoolean(value)) {
        return Boolean(value);
    }
 
    switch (value.toLowerCase()) {
        case "true":
        case "yes":
        case "1":
            return true;
        case "false":
        case "no":
        case "0":
        case null:
            return false;
        default:
            return Boolean(value);
    }
}
